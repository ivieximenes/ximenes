Definição do serviço através do arquivo servicedefinition.xml para que seja registrado no portal.

<?xml version="1.0" encoding="UTF-8" ?>

<serviceDefinition xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance" xsi:noNamespaceSchemaLocation="http://www.lumis.com.br/lumisportal/xsd/servicedefinition.xsd">

 

       <service id="corporateExibeTabela" name="Exibe Tabela" type="lum_service">

              <description>Exemplo de serviço que imprime tabelas na tela</description>

       </service>

 

       <interfaces>

             <interface id="exibeTabela" name="Exibe Tabela" className="corporate.exibetabela.ExibeTabela" />

       </interfaces>

</serviceDefinition>

No servicedefinition.xml nada de diferente é nescessário para que um script funcione.

 

<?xml version="1.0" encoding="UTF-8"?>

<serviceDefinition xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance" xsi:noNamespaceSchemaLocation="http://www.lumis.com.br/lumisportal/xsd/servicedefinition.xsd">

 

 <service name="Script Data Provider Service" type="lum_doui">

         </description>Serviço que exibe informações sobre o próprio serviço e sobre o usuário que o está acessando, mais alguma informação extra, a partir de um Data Provider implementado em Groovy.</description>

 </service>

 

 <interfaces>

         <interface name="Script Data Provider Interface" id="scriptDataProviderInterface" type="lum_doui"/>

 </interfaces>

 

</serviceDefinition>

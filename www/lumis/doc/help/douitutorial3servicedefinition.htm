servicedefinition.xml

Criar o servicedefinition.xml no diretório do serviço, em “lumisdata/def”, por exemplo:"lumisdata/def/Corporate/HoteisAdmin3/servidedefintion.xml”.
 

Abaixo, o conteúdo do arquivo:

<?xml version="1.0" encoding="UTF-8"?>

<serviceDefinition xmlns:xsi= "http://www.w3.org/2001/XMLSchema-instance" xsi:noNamespaceSchemaLocation= "http://www.lumis.com.br/lumisportal/xsd/servicedefinition.xsd">

       <service id="corporateHoteisAdmin3" name="Gerenciamento de Hotéis" type="lum_doui">

             <description>Serviço de Gerenciamento de Hotéis</description>

       </service>

       <interfaces>            

             <!-- Usuário final -->            

             <interface id="list" name="Lista" type="lum_douiList" />            

             <interface id="details" name="Detalhes" type="lum_douiDetails" />            

             <!-- Administração de cidades -->            

             <interface id="adminCidade" name="Administração de cidades" type="lum_douiAdministrationList" />            

             <interface id="addAdminCidade" name="Adicionar cidade" type="lum_douiAdministrationAddPropertyPage">

                    <popupWindowProperties>width=600,height=240</popupWindowProperties>

             </interface>            

             <interface id="editAdminCidade" name="Editar Cidade" type="lum_douiAdministrationEditPropertyPage">

                    <popupWindowProperties>width=600,height=240</popupWindowProperties>

             </interface>                   

             <!-- Administração de hotéis -->

             <interface id="admin" name="Administração" type="lum_douiAdministrationList" />                   

             <interface id="addAdmin" name="Adicionar" type="lum_douiAdministrationAddPropertyPage">

                 <popupWindowProperties>width=600,height=420,left=20,top=20</popupWindowProperties>

             </interface>                   

             <interface id="editAdmin" name="Editar" type="lum_douiAdministrationEditPropertyPage">

                 <popupWindowProperties>width=600,height=420,left=20,top=20</popupWindowProperties>

             </interface>            

       </interfaces>      

</serviceDefinition>

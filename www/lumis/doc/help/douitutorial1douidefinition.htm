douidefinition.xml

 Top  Previous  Next

Criar o douidefinition.xml no diretório do serviço, em “lumisdata/def”, por exemplo: "lumisdata/def/Corporate/HoteisAdmin/douidefinition.xml”.
 

Abaixo, o conteúdo do arquivo:
 

 <?xml version="1.0" encoding="UTF-8"?>

<doui:douiDefinition xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"

  xmlns:doui="http://www.lumis.com.br/lumisportal/xsd/doui"

  xmlns:control= "http://www.lumis.com.br/douicontrols">

 <service id="corporateHoteisAdmin">

  <sources>

   <source id="default" type="table">

    <table>tr3_hotel</table>

    <fields>

     <field id="hotelId" display="false" name="ID" dataType="string" isPrimaryKey="true"/>

     <field id="nome" name="Nome" dataType="string" isPrimaryName="true" required="true" isSearchable="true"/>

     <field id="introducao" name="Introdução" dataType="text" isSearchable="true" isIntroduction="true" display="false"/>

     <field id="descricao" name="Descrição" dataType="html" isSearchable="true" display="false"/>

     <field id="imagem" name="Imagem" dataType="image" doLookup="true"/>

     <field id="lotado" name="Lotado" dataType="integer" display="true">

      <options>

       <option value="0">Não</option>

       <option value="1">Sim</option>

      </options>

     </field>

    </fields>

   </source>

  </sources>

 </service>

 <interfaces>

 

  <!-- Usuário final -->

 

  <interface id="list">

   <sources>

    <source id="default">

     <fields inherit="all">

      <field id="introducao" display="true" />

      <field id="imagem" display="false" />

      <field id="lotado" display="false" />

     </fields>

     <filters>

     <!--  Este filtro escondido, ou oculto, irá exibir as ocorrências que tiverem o campo lotado com valor igual a zero.-->

    

      <filter id="lotado" hidden="true" value="0" />

     </filters>

     <orderBy>

      <!--  O atributo 'direction' define a orientação que será seguida para a exibição dos campos, neste caso é descendente.-->

      <field id="nome" direction="descending" />

     </orderBy>

    </source>

   </sources>

   <hyperLinks>

   <!--  O 'hyperLink' será visualizado na forma de link que apontará para a interface 'details' definida neste arquivo.-->

  

    <hyperLink id="details" interfaceId="details" />

   </hyperLinks>

   <controls>

    <control type="lum_form">   

     <control type="lum_interfaceHeader"/>

     <!-- O controle do tipo 'lum_interfaceHeader' adiciona ao xml gerado um cabeçalho com informações sobre esta interface, 'list',

     contendo nome da interface e do serviço, respectivamente. ex: Lista (Notícias) -->

     <control id="list" type="lum_list" dataId="hotelId" />

     <!-- O controle do tipo 'lum_list' irá adicionar ao xml gerado uma lista de elementos baseado no atributo 'hotelId', como há apenas um

     source, não é necessário explicitá-lo como um atributo do controle.-->

    </control>

   </controls>

  </interface>

 

  <interface id="details">

   <sources>

    <source id="default">

     <fields inherit="all">

      <field id="introducao" display="true" />

      <field id="descricao" display="true" />

      <field id="lotado" display="false" />

     </fields>

    </source>

   </sources>

   <controls>

   <!--O controle do tipo 'lum_autoLayoutDetails' gera automaticamente o layout de interface de detalhes padrão.-->

     <control id="autoLayout" type="lum_autoLayoutDetails"/>

   </controls>

  </interface>

 

  <!-- Administração de hotéis -->

 

  <interface id="admin">

   <sources>

    <source id="default">

     <fields inherit="all">

      <field id="imagem" display="false" />

      <field id="lotado" doOptionLookup="true" />

      <!--  O atributo 'doOptionLookup' habilita/desabilita a opção de exibir o texto no lugar do valor. Por exemplo 'não' em vez de '0'.-->

     

     </fields>

     <orderBy>

      <field id="nome" direction="ascending" />

     </orderBy>

    </source>

   </sources>

   <controls>

    <control type="lum_form">

     <control type="lum_interfaceHeader"/>

     <!--O controle do tipo 'lum_autoAdminListButtons' adiciona os botões padrões da interface de administração, como adicionar, editar e excluir, como neste

     caso os id's das interfaces de adição e edição iniciam respectivamente por add e edit e não há outras interfaces com os mesmo prefixos, os botões

     apontam para essas interfaces sem que seja preciso explicitar.-->

     <control type="lum_autoAdminListButtons"/>  

     <!-- O controle do tipo 'lum_adminList' gera uma lista própria para a interface de administração baseado no 'hotelId', contendo ao lado de cada item, um check box possibilitando que mais de um itens seja selecionado. --> 

     <control type="lum_adminList" id="adminList" dataId="hotelId"/>

    

    </control>

   </controls>

   <processActions>

   <!--O id 'delete' é padrão do processAction de exclusão para os botões gerados automaticamente, acionado através do botão excluir. O tipo 'tableDeleteData' indica que o(s)

    elemento(s) selecionado(s) será(ão) apagado(s) da tabela que é identificada através do 'sourceId' definido -->

    <processAction id="delete" type="tableDeleteData" sourceId="default">

    <!--O 'response' do tipo 'doui_standardEmbedded' define que o comportamento do processAction, a resposta, será padrão para interfaces instanciada em páginas

      comuns do portal, que não são popup's.-->

     <response type="doui_standardEmbedded"/>

    </processAction>

   </processActions>

  </interface>

 

  <interface id="addAdmin">

   <sources>

   <!--  O atributo 'readData="never"' define que os dados do source não serão carregados.-->

    <source id="default" readData="never"/>

   </sources>

   <controls>

    <control type="lum_form">

     <control type="lum_propertyPage">

      <control type="lum_controlGroup" title="Informações Básicas">

       <control type="lum_table">

        <tr>

         <td>

          <control type="lum_label" dataId="nome"/>

         </td>

         <td>

          <control type="lum_inputText" id="nome"/>

         </td>

        </tr>

        <tr>

         <td>

          <control type="lum_label" dataId="introducao"/>

         </td>

         <td>

          <control type="lum_inputTextArea" id="introducao" rows="2"/>

         </td>

        </tr>

        <tr>

         <td colspan="2">

          <control type="lum_label" dataId="descricao"/>

         </td>

        </tr>

        <tr>

         <td colspan="2">

          <control type="lum_inputHtmlEditor" id="descricao"/>

         </td>

        </tr>

        <tr>

         <td>

          <control type="lum_label" dataId="imagem"/>

         </td>

         <td>

          <control type="lum_imagePicker" id="imagem"/>

         </td>

        </tr>

        <tr>

         <td>

          <control type="lum_label" dataId="lotado"/>

         </td>

         <td>

          <control type="lum_checkBox" id="lotado"/>

         </td>

        </tr>

       </control>

      </control>

     </control>

    </control>

   </controls>

   <processActions>

   <!-- O id 'commit' é padrão do processAction para o botão 'Ok' gerado automaticamente. O tipo 'tableAddData' indica que o elemento selecionado será

    inserido na tabela que é identificada através do 'sourceId' definido-->

    <processAction id="commit" type="tableAddData" sourceId="default">

     <response type="doui_standardPopup" />

    </processAction>

   </processActions>

  </interface>

 

  <interface id="editAdmin">

   <sources>

    <source id="default">

     <filters>

     <!-- O atributo 'required' define se o preenchimento deste campo é obrigatório, ou não. Quando criamos uma tabela e definimos que um

      determinado campo não pode ser nulo, o preenchimento deste campo passa a ser obrigatório, 'required="true"' por default implicitamente,

      nos controles de entrada de dados, como por exemplo 'lum_inputText', mas quando definimos filtros, se desejamos que o seu preenchimento

      seja obrigatório, é preciso definir através deste atributo.-->

      <filter id="hotelId" required="true" hidden="true" />

     </filters>

    </source>

   </sources>

   <controls>

    <control type="lum_form">

     <control type="lum_filters" id="filters" />

     <control type="lum_propertyPage">

      <control type="lum_controlGroup" title="Informa&#231;&#245;es B&#225;sicas">

       <control type="lum_table">

        <tr>

         <td>

          <control type="lum_label" dataId="nome"/>

         </td>

         <td>

          <control type="lum_inputText" id="nome"/>

         </td>

        </tr>

        <tr>

         <td>

          <control type="lum_label" dataId="introducao"/>

         </td>

         <td>

          <control type="lum_inputTextArea" id="introducao" rows="2"/>

         </td>

        </tr>

        <tr>

         <td colspan="2">

          <control type="lum_label" dataId="descricao"/>

         </td>

        </tr>

        <tr>

         <td colspan="2">

          <control type="lum_inputHtmlEditor" id="descricao"/>

         </td>

        </tr>

        <tr>

         <td>

          <control type="lum_label" dataId="imagem"/>

         </td>

         <td>

          <control type="lum_imagePicker" id="imagem"/>

         </td>

        </tr>

        <tr>

         <td>

          <control type="lum_label" dataId="lotado"/>

         </td>

         <td>

          <control type="lum_checkBox" id="lotado"/>

         </td>

        </tr>

       </control>

      </control>

     </control>

    </control>

   </controls>

   <processActions>

   <!--O id 'commit' é padrão do processAction para o botão 'Ok' gerado automaticamente, acionado através do botão 'Ok'. O tipo 'tableUpdateData' indica que o elemento selecionado será

    atualisado na tabela que é identificada através do 'sourceId' definido, 'default' no caso.-->

    <processAction id="commit" type="tableUpdateData" sourceId="default">

     <response type="doui_standardPopup" />

    </processAction>

   </processActions>

  </interface>

 </interfaces>

</doui:douiDefinition>

Nota:

•Caso o banco utilizado seja Oracle, campos declarados como “doLookup” devem ter no máximo 27 caracteres para o seu identificador. Isso porque, o sufixo utilizado internamente para campo possui 3 caracteres. Da mesma forma, a quantidade de caracteres de filtro que possuem lookpup deve ser de 24 caracteres e de controles que possuem lookup devem ser de 27 caracteres.

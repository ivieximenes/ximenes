No banco de dados do projeto do portal, criar as seguintes tabelas cujos campos serão utilizados pelos sources do douidefinition.xml:
 

CREATE TABLE tr3_hotel

(

 hotelId   CHAR(32) PRIMARY KEY NOT NULL,

 nome   NVARCHAR(100) NOT NULL,

 introducao  NTEXT NULL,

 descricao  NTEXT NOT NULL,

 imagem   CHAR(32) NULL,

 lotado   INT NOT NULL DEFAULT 0,

 cidadeId  CHAR(32) NULL

)

CREATE TABLE tr3_cidade

(

 cidadeId  CHAR(32) PRIMARY KEY NOT NULL,

 nome   NVARCHAR(100) NOT NULL,

 uf    CHAR(2) NOT NULL

)
